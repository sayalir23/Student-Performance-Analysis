CREATE TABLE marks (
student_id INT,
subject VARCHAR(50),
marks INT
);

##Total Marks per Student
SELECT student_id, SUM(marks) AS total_marks
FROM marks
GROUP BY student_id;

##Calculate average marks per students
SELECT student_id, AVG(marks) as avg_marks
FROM marks
GROUP BY student_id; 

##show top performing students
SELECT student_id, SUM(marks) AS total_marks
FROM marks
GROUP BY student_id
ORDER BY total_marks DESC
LIMIT 1;

##show highest marks in each subject
SELECT subject , MAX(marks) as highest_marks
from marks
GROUP BY subject;
